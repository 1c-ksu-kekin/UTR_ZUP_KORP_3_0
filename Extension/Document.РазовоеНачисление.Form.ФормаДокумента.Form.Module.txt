Процедура ЮТР_НачисленияСотрудникПриИзменении(Элемент)
	
	ТекущиеДанные = Элементы.Начисления.ТекущиеДанные;
	
	Если Не (ТекущиеДанные = Неопределено) Тогда 
		Если ЗначениеЗаполнено(ТекущиеДанные.Сотрудник) Тогда
			Если не ЗначениеЗаполнено(ТекущиеДанные.ЮТР_Дата) Тогда
				ТекущиеДанные.ЮТР_Дата = ТекущаяДата();
			КонецЕсли;
			Если не ЗначениеЗАполнено(ТекущиеДанные.ЮТР_Основание) Тогда 
				ТекущиеДанные.ЮТР_Основание = "Заявление, трудовая книжка, решение органа службы занятости населения.";
			КонецЕсли 	  
		Иначе
			ТекущиеДанные.ЮТР_Дата = "";
			ТекущиеДанные.ЮТР_Основание = "";
		КонецЕсли
	КонецЕсли
