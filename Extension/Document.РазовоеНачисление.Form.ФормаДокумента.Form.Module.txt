&НаКлиенте
Процедура ЮТР_НачисленияСотрудникПриИзменении(Элемент)
	
	ТекущиеДанные = Элементы.Начисления.ТекущиеДанные;
	
	Если Не (ТекущиеДанные = Неопределено) Тогда 
		Если ЗначениеЗаполнено(ТекущиеДанные.Сотрудник) Тогда
			Если не ЗначениеЗаполнено(ТекущиеДанные.ЮТР_Дата) Тогда
				ТекущиеДанные.ЮТР_Дата = ТекущаяДата();
			КонецЕсли;
			Если не ЗначениеЗАполнено(ТекущиеДанные.ЮТР_Основание) Тогда 
				ТекущиеДанные.ЮТР_Основание = "Заявление, трудовая книжка, решение органа службы занятости населения.";
			КонецЕсли 	  
		Иначе
			ТекущиеДанные.ЮТР_Дата = "";
			ТекущиеДанные.ЮТР_Основание = "";
		КонецЕсли
	КонецЕсли	 
	
КонецПроцедуры
#КонецОбласти

&НаКлиенте
Процедура ЮТР_ПриНажатииПолеРаспределениеПоТерриториямУсловиямТруда(ОписаниеТаблицы, Элемент, ВыбраннаяСтрока, Поле, СтандартнаяОбработка)
	
	ПараметрыВыбора1 = РасчетЗарплатыРасширенныйКлиент.ПараметрыВыбораПолеРаспределениеПоТерриториямУсловиямТруда();
	ПараметрыВыбора1.Форма = ЭтаФорма;
	ПараметрыВыбора1.ОписаниеДокумента = ОписаниеДокумента(ЭтаФорма);
	ПараметрыВыбора1.ОписаниеТаблицы = ОписаниеТаблицы;
	
	РасчетЗарплатыРасширенныйКлиент.ПриНажатииПолеРаспределениеПоТерриториямУсловиямТруда(Элемент, ВыбраннаяСтрока, Поле, СтандартнаяОбработка, ПараметрыВыбора1);
	
КонецПроцедуры

&НаКлиенте
Процедура ЮТР_НачисленияВыбор(Элемент, ВыбраннаяСтрока, Поле, СтандартнаяОбработка)
	
	ЮТР_ПриНажатииПолеРаспределениеПоТерриториямУсловиямТруда(ОписаниеТаблицыНачислений(), Элемент, ВыбраннаяСтрока, Поле, СтандартнаяОбработка);	
	
	Если Не СтандартнаяОбработка Тогда
		Возврат;
	КонецЕсли;
	
	ДополнительныеПараметры = Новый Структура;
	ДополнительныеПараметры.Вставить("Элемент", Элемент);
	ДополнительныеПараметры.Вставить("ВыбраннаяСтрока", ВыбраннаяСтрока);
	ДополнительныеПараметры.Вставить("Поле", Поле);
	
	Оповещение = Новый ОписаниеОповещения("НачисленияВыборЗавершение", ЭтотОбъект, ДополнительныеПараметры);
	РасчетЗарплатыРасширенныйКлиент.ВыполнитьКомандуРасшифровкиЗначенияПоказателя(ЭтаФорма, ОписаниеТаблицыНачислений(), Элемент, ВыбраннаяСтрока, Поле, СтандартнаяОбработка, Оповещение);
	
	Если Найти(Поле.Имя, "КомандаРедактированияРаспределения") <> 0 Тогда
		СтандартнаяОбработка = Ложь;
	КонецЕсли;
	
	УстановитьВыполнениеОбработчиковСобытия(Ложь);
	
КонецПроцедуры
