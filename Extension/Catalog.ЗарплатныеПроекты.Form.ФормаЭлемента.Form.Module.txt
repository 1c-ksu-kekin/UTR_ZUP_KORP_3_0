#Область СлужебныеПроцедурыИФункции

 &НаКлиентеНаСервереБезКонтекста
Процедура ЮТР_УстановитьДоступностьЭлементовФормы(Форма, ИспользоватьЭлектронныйДокументооборотСБанком)

	ОбщегоНазначенияКлиентСервер.УстановитьСвойствоЭлементаФормы(
		Форма.Элементы, "ЮТР_ИспользоватьВыгрузкуТХТ", "Доступность", ИспользоватьЭлектронныйДокументооборотСБанком);
	ОбщегоНазначенияКлиентСервер.УстановитьСвойствоЭлементаФормы(
		Форма.Элементы, "ЮТР_ИспользоватьРазделениеПоПризнакуРезидент", "Доступность", ИспользоватьЭлектронныйДокументооборотСБанком);
	ОбщегоНазначенияКлиентСервер.УстановитьСвойствоЭлементаФормы(
		Форма.Элементы, "ЮТР_МаскаЛицевогоСчетаНерезидента", "Доступность", ИспользоватьЭлектронныйДокументооборотСБанком);		
	
КонецПроцедуры

&НаКлиенте
Процедура ЮТР_ЮТР_ИспользоватьРазделениеПоПризнакуРезидентПриИзменении(Элемент)
	ЭтаФорма.Элементы.ЮТР_МаскаЛицевогоСчетаНерезидента.Видимость = Объект.ЮТР_ИспользоватьРазделениеПоПризнакуРезидент;
КонецПроцедуры

&НаКлиенте
Процедура ЮТР_ПриОткрытии(Отказ)
	ЭтаФорма.Элементы.ЮТР_МаскаЛицевогоСчетаНерезидента.Видимость = Объект.ЮТР_ИспользоватьРазделениеПоПризнакуРезидент;
