	Если ЗначениеЗаполнено(Объект.ЮТР_ДатаРожденияРебенка) Тогда
		
		Объект.ДатаОкончания = КонецДня(ДобавитьМесяц(Объект.ЮТР_ДатаРожденияРебенка, 36));
		
		Если Объект.ВыплачиватьПособиеДоПолутораЛет Тогда 
			Объект.ДатаОкончанияПособияДоПолутораЛет = КонецДня(ДобавитьМесяц(Объект.ЮТР_ДатаРожденияРебенка, 18));
		КонецЕсли;
		
		Если Объект.ВыплачиватьПособиеДоТрехЛет Тогда 
			Объект.ДатаОкончанияПособияДоТрехЛет = КонецДня(ДобавитьМесяц(Объект.ЮТР_ДатаРожденияРебенка, 36));
		КонецЕсли;
		
	КонецЕсли;
